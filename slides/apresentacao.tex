\begin{frame}{Introdução}
\begin{columns}[T]
\begin{column}{0.58\textwidth}
\textbf{Conjunto de Mandelbrot} (imagem 7680$\times$4320) em versão \textbf{híbrida MPI + OpenMP}, com \textbf{dois níveis de paralelismo}:
\begin{itemize}
  \item \textbf{Entre nós (MPI)} --- modelo \emph{Coordenador/Trabalhador}: o coordenador distribui blocos de linhas sob demanda; \textbf{1 processo pesado por nó}.
  \item \textbf{Dentro do nó (OpenMP)} --- \emph{workpool sem mestre}: todas as threads pegam a próxima linha de uma \textbf{estrutura de dados} compartilhada (\texttt{atomic capture}), sem thread mestre.
\end{itemize}
\vspace{3pt}
Carga muito \textbf{irregular} (pontos internos custam \texttt{max\_iter}, externos escapam cedo) $\Rightarrow$ o workpool faz o \textbf{balanceamento dinâmico}.
\vspace{3pt}

Executado em \textbf{4 nós} do cluster Atlantica; \texttt{OMP\_NUM\_THREADS} de 1 a 16 (2 threads/núcleo, HT). Saída \textbf{byte-idêntica} à sequencial.
\end{column}

% ===================== SLIDE 3 -- CONCLUSAO =====================
\begin{frame}{Conclusões}
\begin{itemize}
  \item Cumprimos o modelo: \textbf{MPI Coordenador/Trabalhador} (1 processo pesado/nó) $+$ \textbf{workpool OpenMP sem mestre} (controle por estrutura de dados, 2 threads/núcleo com HT).
  \item \textbf{Boa escalabilidade}: speed-up de \textbf{46$\times$} (forte) e \textbf{50$\times$} (fraca) em 64 unidades; escalabilidade fraca quase ideal.
  \item A queda de eficiência em 64 unidades é o \textbf{limite do Hyper-Threading} (2 fluxos/núcleo), não do algoritmo; no início há \emph{superlinearidade} por melhor uso de \textbf{cache}.
  \item \textbf{Alocação do coordenador}: vale dedicar \textbf{1 núcleo} a ele ($-16\%$, evita oversubscrição), mas \textbf{nunca um nó} inteiro ($+63\%$).
  \item \textbf{Híbrido $\times$ MPI pura}: desempenho \textbf{praticamente igual} (curvas coincidentes); o híbrido usa \textbf{muito menos ranks} MPI (5 vs 64) $\Rightarrow$ menos memória e mensagens, vantagem que cresce com a escala.
\end{itemize}
